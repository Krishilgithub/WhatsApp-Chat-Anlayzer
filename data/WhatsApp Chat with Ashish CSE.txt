    private final int[] buffer;
    private int count = 0;
    private final int capacity;

    public ProducerConsumerWithArray(int size) {
        buffer = new int[size];
        capacity = size;
    }

    // Producer method
    public void produce() throws InterruptedException {
        int value = 0;
        while (true) {
            synchronized (this) {
                // If buffer is full, producer will wait
                while (count == capacity) {
                    wait();
                }
                
                // Add produced item to buffer
                buffer[count] = value;
                System.out.println("Produced: " + value);
                value++;
                count++;

                // Notify the consumer that an item is available
                notify();
            }
            // Simulate some delay in production
            Thread.sleep(500);
        }
    }

    // Consumer method
    public void consume() throws InterruptedException {
        while (true) {
            synchronized (this) {
                // If buffer is empty, consumer will wait
                while (count == 0) {
                    wait();
                }

                // Consume an item from buffer
                int value = buffer[count - 1];
                System.out.println("Consumed: " + value);
                count--;

                // Notify the producer that space is available
                notify();
            }
            // Simulate some delay in consumption
            Thread.sleep(500);
        }
    }

    public static void main(String[] args) throws InterruptedException {
        ProducerConsumerWithArray pc = new ProducerConsumerWithArray(5); // buffer size = 5

        // Create producer thread
        Thread producerThread = new Thread(() -> {
            try {
                pc.produce();
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        });

        // Create consumer thread
        Thread consumerThread = new Thread(() -> {
            try {
                pc.consume();
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        });

        // Start both threads
        producerThread.start();
        consumerThread.start();

        // Wait for both threads to finish
        producerThread.join();
        consumerThread.join();
    }
}

void linearSearch(int arr[], int n, int x, int &position, int &comparisons) {
    comparisons = 0;
    position = -1;
    for (int i = 0; i < n; ++i) {
        comparisons++;
        if (arr[i] == x) {
            position = i + 1;
            return;
        }
    }
}

void binarySearch(int arr[], int n, int x, int &position, int &comparisons) {
    comparisons = 0;
    position = -1;
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = (right + left) / 2;
        comparisons++;
        if (arr[mid] == x) {
            position = mid + 1;
            right = mid - 1;
        } else if (arr[mid] < x) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
}
